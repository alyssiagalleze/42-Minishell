-chdir :
	prototype	:int	chdir(const char *path);
	return		:0 on success, -1 on error (errno)

	remplace le repertoire de travail courant du processus par celui indique dans <path>

-stat :
	prototype	:int	stat(const char *restrict path, struct stat *restrict statbuf);
	return		:0 on success, -1 on error (errno)

	returns information about a file in the buffer pointed to by statbuf.
	permissions are required on the path but not on the file itself
-lstat :
	prototype	:int	lstat(const char *restrict path, struct stat *restrict statbuf);
	return		:0 on success, -1 on error (errno)

	similar to stat, except that if path is a symbolic link, it returns info about link itself (not the file referred to)

-fstat :
	prototype	:int	fstat(int fd, struct stat *restrict statbuf)
	return		:0 on success, -1 on error (errno)

	same as stat, but the file is retrieved via it's file descriptor

-unlink :
	prototype	:int	unlink(const char *pathname);
	return		:0 on success, -1 on error (errno)

	destroys a name on a file system, 
	if it was the last link to a file and no process has opened it, it is deleted
	if it was the last link to a file but a process keeps it opened, it stays until the last referenced file descriptor is closedir
	if the name is a symlink, it is destroyed

-execve :
	prototype	:int	execve(const char *path, char **const _Nullable argv, char **const _Nullable envp);
	return		:no return on success, -1 on error

	executes program refered to by path, 
	replaces the program that is currently being run by the calling process by a new program (initialized stack, heap and data segments)

-opendir :
	prototype	:DIR	*opendir(const char *name);
	return		:pointer to the directory stream, NULL on error(errno)

	opens a directory stream (usefull to read its content) corresponding to the directory name, it is positionned at the first entry in the directory

-readdir :
	prototype	:struct dirent *readdir(DIR *dirp);
	return		:pointer to a dirent structure 
				if the end of the directory stream is reached, NULL is returned(errno unchanged)
				on error, NULL and errno set 

	returns a pointer to a dirent structure representing the next directory entry in the directory stream.

	example		:
	#include <stdio.h>
    #include <dirent.h>

    int main (int argc, char *argv[]) {
        struct dirent *pDirent;
        DIR *pDir;

        // Ensure correct argument count.

        if (argc != 2) {
            printf ("Usage: testprog <dirname>\n");
            return 1;
        }

        // Ensure we can open directory.

        pDir = opendir (argv[1]);
        if (pDir == NULL) {
            printf ("Cannot open directory '%s'\n", argv[1]);
            return 1;
        }

        // Process each entry.

        while ((pDirent = readdir(pDir)) != NULL) {
            printf ("[%s]\n", pDirent->d_name);
        }

        // Close directory and exit.

        closedir (pDir);
        return 0;
    }

-tgetent :
	prototype	:int tgetent(char *bp, const char *name);
	return		:1 entry found and loaded, 0 no entry, -1 data base not found

	loads the database of name terminal in a memory field so we can get information through other functions
	such as (tgetstr, tgetnum, tgetflag)

-tgetflag :
	prototype	:int	tgetflag(const char *id);
	return		:1 capacity exists, 0 capacity is not defined

	test a boolean capacity (termcap/terminfo base) of the terminal (ex : does the terminal automatically adds a newline ?)

-tgetnum :
	prototype	:int	tgetnum(char *id)
	return		:int or -1 on error

	gets a numeric capacity (termcap/terminfo base) (ex : nb of columns, lines, paddind time etc)

-tgetstr :
	prototype	:char	*tgetstr(char *id, char **area)
	return		:pointer to the chain containing the control sequence, NULL if capacity is not available

	allows to get a capacity of string type, these are escape sequences to send the terminal to make an action.
	area is NULLABLE, if NULL ncurses manages memory otherwise the string is copied in this memory area and *area is moved forward
	
	example :

	#include <stdio.h>
	#include <stdlib.h>
	#include <curses.h>
	#include <term.h>

	int main(void) {
    	char *term = getenv("TERM");
    	if (!term) {
        	fprintf(stderr, "La variable TERM n'est pas définie.\n");
        	return 1;
    	}

    	if (tgetent(NULL, term) != 1) {
        	fprintf(stderr, "Impossible de charger la base pour %s\n", term);
        	return 1;
    	}

    	// Récupérer la séquence "clear screen"
    	char *clear = tgetstr("cl", NULL);
    	if (clear) {
        	printf("%s", clear); // Écrit la séquence → efface l’écran
    	} else {
        	fprintf(stderr, "Le terminal ne supporte pas 'clear screen'.\n");
    	}	

    	return 0;
	}

-tgoto : 
	prototype	:char	*tgoto(const char *cap, int col, int row)
	return		:

	some string capacities use parameters for example : cm (cursor motion) needs two values, column and line, 
	tgoto takes this string and turns it into a real sequence ready to be sent to a terminal

	example :

	#include <stdio.h>
	#include <stdlib.h>
	#include <curses.h>
	#include <term.h>

	int main(void) {
    	char *term = getenv("TERM");
    	if (!term) {
        	fprintf(stderr, "La variable TERM n'est pas définie.\n");
        	return 1;
    	}

    	if (tgetent(NULL, term) != 1) {
        	fprintf(stderr, "Impossible de charger la base pour %s\n", term);
        	return 1;
    	}

    	// Récupérer la capacité "cursor motion"
    	char *cm = tgetstr("cm", NULL);
    	if (!cm) {
        	fprintf(stderr, "Le terminal ne supporte pas 'cursor motion'.\n");
        	return 1;
    	}

    	// Construire la séquence pour aller à (ligne=10, col=20)
    	char *seq = tgoto(cm, 20, 10);
    	printf("%sHello ici !\n", seq);

    	return 0;
	}

-tputs :
	prototype	:int	tputs(const char *str, int affcnt, int (*putc)(int));
	return		:0 on success, ERR otherwise
					(most of the time, the return value isn't checked bc tputs calls directly the putc function)

	sends a control sequence to the terminal(obtained with tgetstr or generated with tgoto), 
	manages padding breaks(old constraints limnked to the efficiency of the terminal, rarely usefull nowadays)
	equivalent to "printf("%s", str)" that respects termcap conventions

	example :

    if (cm) {
        char *seq = tgoto(cm, 20, 10); // colonne=20, ligne=10
        tputs(seq, 1, putchar);
        printf("Hello ici !");
    }
